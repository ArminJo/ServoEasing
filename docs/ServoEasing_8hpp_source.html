2  * ServoEasing.hpp
3  *
4  * Enables smooth movement from one servo position to another.
5  * Linear as well as other ease movements (e.g. cubic) for all servos attached to the Arduino Servo library are provided.
6  * Interface is in degree but internally only microseconds (if using Servo library) or units (if using PCA9685 expander) are used,
7  * since the resolution is better and we avoid the map function on every Servo.write().

9  *

514 uint8_t ServoEasing::attach(int aPin, int aMicrosecondsForServoLowDegree, int aMicrosecondsForServoHighDegree, int aServoLowDegree,
515  int aServoHighDegree) {
516  /*
517  * Get the 0 and 180 degree values.
518  */
519  int tMicrosecondsForServo0Degree = map(0, aServoLowDegree, aServoHighDegree, aMicrosecondsForServoLowDegree,
520  aMicrosecondsForServoHighDegree);
521  int tMicrosecondsForServo180Degree = map(180, aServoLowDegree, aServoHighDegree, aMicrosecondsForServoLowDegree,
522  aMicrosecondsForServoHighDegree);
523 
524  mServoPin = aPin;
525 #if defined(USE_PCA9685_SERVO_EXPANDER)
526 # if defined(USE_SERVO_LIB)
527  if (mServoIsConnectedToExpander) {
528  // set units
529  mServo0DegreeMicrosecondsOrUnits = MicrosecondsToPCA9685Units(tMicrosecondsForServo0Degree);
530  mServo180DegreeMicrosecondsOrUnits = MicrosecondsToPCA9685Units(tMicrosecondsForServo180Degree);
531  } else {
532  // set microseconds
533  mServo0DegreeMicrosecondsOrUnits = tMicrosecondsForServo0Degree;
534  mServo180DegreeMicrosecondsOrUnits = tMicrosecondsForServo180Degree;
535  }
536 # else
537  // set units
538  mServo0DegreeMicrosecondsOrUnits = MicrosecondsToPCA9685Units(tMicrosecondsForServo0Degree);
539  mServo180DegreeMicrosecondsOrUnits = MicrosecondsToPCA9685Units(tMicrosecondsForServo180Degree);
540 # endif
541 #else
542  // set microseconds
543  mServo0DegreeMicrosecondsOrUnits = tMicrosecondsForServo0Degree;
544  mServo180DegreeMicrosecondsOrUnits = tMicrosecondsForServo180Degree;
545 #endif
546 
547  /*
548  * Now put this servo instance into list of servos
549  */
550  uint8_t tReturnValue = INVALID_SERVO; // flag indicating an invalid servo index

1430 #endif
1431 
1432  uint32_t tMillisSinceStart = millis() - mMillisAtStartMove;
1433  if (tMillisSinceStart >= mMillisForCompleteMove) {

1442  }

1445  if (mEasingType == EASE_LINEAR) {
1446  /*
1447  * Use faster non float arithmetic
1448  * Linear movement: new position is: start position + total delta * (millis_done / millis_total aka "percentage of completion")
1449  * 40 us to compute
1450  * Cast to int32 required for mMillisForCompleteMove for 32 bit platforms, otherwise we divide signed by unsigned. Thanks to drifkind.

1456  * Non linear movement -> use floats
1457  * Compute tPercentageOfCompletion - from 0.0 to 1.0
1458  * The expected result of easing function is from 0.0 to 1.0
1459  * or from EASE_FUNCTION_MICROSECONDS_INDICATOR_OFFSET for direct microseconds result
1460  */

1462  float tFactorOfMovementCompletion = 0.0;
1463

1467  // Use IN function direct: Call with PercentageOfCompletion | 0.0 to 1.0. FactorOfMovementCompletion is returnValue (from 0.0 to 1.0)
1468  tFactorOfMovementCompletion = callEasingFunction(tFactorOfTimeCompletion);
1469 
1470  } else if (tCallStyle == CALL_STYLE_OUT) {

1479  }

1484  } else {

1508 # if defined(USE_SERVO_LIB)

1513  }
1514 # else
1515  tNewMicrosecondsOrUnits = MicrosecondsToPCA9685Units(tFactorOfMovementCompletion);
1516 # endif
1517 #else
1518  tNewMicrosecondsOrUnits = tFactorOfMovementCompletion; // get the microseconds delivered by the function
1519 #endif
1520  } else
1521 #endif
1522 #if defined(ENABLE_EASE_USER)
1523  // check for degree values from -180 to 180 (tFactorOfMovementCompletion from 20 to 380)
1524  if (tFactorOfMovementCompletion >= EASE_FUNCTION_DEGREE_THRESHOLD) {
1525  // Here we have called an easing function, which returns degree instead of the factor of completion (0.0 to 1.0)
1526  tNewMicrosecondsOrUnits = DegreeOrMicrosecondToMicrosecondsOrUnits(
1527  (float) (tFactorOfMovementCompletion - EASE_FUNCTION_DEGREE_INDICATOR_OFFSET + 0.5));
1528  } else
1529 #endif
