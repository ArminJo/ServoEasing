31 /* In order to use the library, you need to define SDA_PIN, SCL_PIN,
32  * SDA_PORT and SCL_PORT before including this file. Have a look at
33  * http://www.arduino.cc/en/Reference/PortManipulation for finding out
34  * which values to use. For example, if you use digital pin 3 (corresponding
35  * to PD3) for SDA and digital pin 13 (corresponding to PB5)
36  * for SCL on a standard Arduino,
37  * you have to use the following definitions:
38  * #define SDA_PIN 3
39  * #define SDA_PORT PORTD
40  * #define SCL_PIN 5
41  * #define SCL_PORT PORTB
42  *
43  * Alternatively, you can define the compile time constant I2C_HARDWARE,
44  * in which case the TWI hardware is used. In this case you have to use
45  * the standard SDA/SCL pins (and, of course, the chip needs to support
46  * this).
47  *
48  * You can also define the following constants (see also below):
49  ' - I2C_PULLUP = 1 meaning that internal pullups should be used
50  * - I2C_CPUFREQ, when changing CPU clock frequency dynamically
51  * - I2C_FASTMODE = 1 meaning that the I2C bus allows speeds up to 400 kHz
52  * - I2C_SLOWMODE = 1 meaning that the I2C bus will allow only up to 25 kHz
53  * - I2C_NOINTERRUPT = 1 in order to prohibit interrupts while
54  * communicating (see below). This can be useful if you use the library
55  * for communicating with SMbus devices, which have timeouts.
56  * Note, however, that interrupts are disabled from issuing a start condition
57  * until issuing a stop condition. So use this option with care!
58  * - I2C_TIMEOUT = 0...10000 msec in order to return from the I2C functions
59  * in case of a I2C bus lockup (i.e., SCL constantly low). 0 means no timeout.
60  * - I2C_MAXWAIT = 0...32767 number of retries in i2c_start_wait. 0 means never stop.
61  */

282 void i2c_delay_half(void) { // function call 3 cycles => 3C
283 #if I2C_DELAY_COUNTER < 1
284  __asm__ __volatile__ (" ret");
285  // 7 cycles for call and return
286 #else
287  __asm__ __volatile__
288  (
289  " ldi r25, %[DELAY] ;load delay constant ;; 4C \n\t"
290  "_Lidelay: \n\t"
291  " dec r25 ;decrement counter ;; 4C+xC \n\t"
292  " brne _Lidelay ;;5C+(x-1)2C+xC\n\t"
293  " ret ;; 9C+(x-1)2C+xC = 7C+xC"
294  : : [DELAY] "M" I2C_DELAY_COUNTER : "r25");
295  // 7 cycles + 3 times x cycles
296 #endif
297 }

299 void i2c_wait_scl_high(void) {
300 #if I2C_TIMEOUT <= 0
301  __asm__ __volatile__
302  ("_Li2c_wait_stretch: \n\t"
303  " sbis %[SCLIN],%[SCLPIN] ;wait for SCL high \n\t"
304 #if __AVR_HAVE_JMP_CALL__
305  " jmp _Li2c_wait_stretch \n\t"
306 #else
307  " rjmp _Li2c_wait_stretch \n\t"
308 #endif
309  " cln ;signal: no timeout \n\t"
310  " ret "
311  : : [SCLIN] "I" (SCL_IN), [SCLPIN] "I" (SCL_PIN));
312 #else // I2C_TIMEOUT <= 0
313  __asm__ __volatile__
314  ( " ldi r27, %[HISTRETCH] ;load delay counter \n\t"
315  " ldi r26, %[LOSTRETCH] \n\t"
316  "_Lwait_stretch: \n\t"
317  " clr __tmp_reg__ ;do next loop 255 times \n\t"
318  "_Lwait_stretch_inner_loop: \n\t"
319 #if __AVR_HAVE_JMP_CALL__
320  " call _Lcheck_scl_level ;call check function ;; 12C \n\t"
321 #else
322  " rcall _Lcheck_scl_level ;call check function ;; 12C \n\t"
323 #endif
324  " brpl _Lstretch_done ;done if N=0 ;; +1 = 13C\n\t"
325  " dec __tmp_reg__ ;dec inner loop counter;; +1 = 14C\n\t"
326  " brne _Lwait_stretch_inner_loop ;; +2 = 16C\n\t"
327  " sbiw r26,1 ;dec outer loop counter \n\t"
328  " brne _Lwait_stretch ;continue with outer loop \n\t"
329  " sen ;timeout -> set N-bit=1 \n\t"
330 #if __AVR_HAVE_JMP_CALL__
331  " jmp _Lwait_return ;and return with N=1\n\t"
332 #else
333  " rjmp _Lwait_return ;and return with N=1\n\t"
334 #endif
335  "_Lstretch_done: ;SCL=1 sensed \n\t"
336  " cln ;OK -> clear N-bit \n\t"
337 #if __AVR_HAVE_JMP_CALL__
338  " jmp _Lwait_return ; and return with N=0 \n\t"
339 #else
340  " rjmp _Lwait_return ; and return with N=0 \n\t"
341 #endif
342  "_Lcheck_scl_level: ;; call = 3C\n\t"
343  " cln ;; +1C = 4C \n\t"
344  " sbic %[SCLIN],%[SCLPIN] ;skip if SCL still low ;; +2C = 6C \n\t"
345 #if __AVR_HAVE_JMP_CALL__
346  " jmp _Lscl_high ;; +0C = 6C \n\t"
347 #else
348  " rjmp _Lscl_high ;; +0C = 6C \n\t"
349 #endif
350  " sen ;; +1 = 7C\n\t "
351  "_Lscl_high: "
352  " nop ;; +1C = 8C \n\t"
353  " ret ;return N-Bit=1 if low ;; +4 = 12C\n\t"
354 
355  "_Lwait_return:"
356  : : [SCLIN] "I" (SCL_IN), [SCLPIN] "I" (SCL_PIN),
357  [HISTRETCH] "M" (I2C_MAX_STRETCH>>8),
358  [LOSTRETCH] "M" (I2C_MAX_STRETCH&0xFF)
359  : "r26", "r27");
360 #endif // I2C_TIMEOUT <= 0
361 }
362 #endif // !I2C_HARDWARE

364 bool i2c_init(void)
365 #if I2C_HARDWARE
366 {
367 #if __has_include("digitalWriteFast.h")
368 #include "digitalWriteFast.h"
369 # if I2C_PULLUP
370  digitalWriteFast(SDA, 1);
371  digitalWriteFast(SCL, 1);
372 # else
373  digitalWriteFast(SDA, 0);
374  digitalWriteFast(SCL, 0);
375 # endif
376 #else // __has_include("digitalWriteFast.h")
377 # if I2C_PULLUP
378  digitalWrite(SDA, 1);
379  digitalWrite(SCL, 1);
380 # else
381  digitalWrite(SDA, 0);
382  digitalWrite(SCL, 0);
383 # endif
384 #endif // __has_include("digitalWriteFast.h")
385 #if ((I2C_CPUFREQ/SCL_CLOCK)-16)/2 < 250
386  TWSR = 0; /* no prescaler */
387  TWBR = ((I2C_CPUFREQ/SCL_CLOCK)-16)/2; /* must be > 10 for stable operation */
388 #else
389  TWSR = (1<<TWPS0); // prescaler is 4
390  TWBR = ((I2C_CPUFREQ/SCL_CLOCK)-16)/8;
391 #endif
392  return (digitalRead(SDA) != 0 && digitalRead(SCL) != 0);
393 }
394 #else // I2C_HARDWARE
395  {
396  __asm__ __volatile__
397  (" cbi %[SDADDR],%[SDAPIN] ;release SDA \n\t"
398  " cbi %[SCLDDR],%[SCLPIN] ;release SCL \n\t"
399 #if I2C_PULLUP
400  " sbi %[SDAOUT],%[SDAPIN] ;enable SDA pull-up\n\t"
401 #else
402  " cbi %[SDAOUT],%[SDAPIN] ;clear SDA output value \n\t"
403 #endif
404 #if I2C_PULLUP
405  " sbi %[SCLOUT],%[SCLPIN] ;enable SCL pull-up\n\t"
406 #else
407  " cbi %[SCLOUT],%[SCLPIN] ;clear SCL output value \n\t"
408 #endif
409  " clr r24 ;set return value to false \n\t"
410  " clr r25 ;set return value to false \n\t"
411  " sbis %[SDAIN],%[SDAPIN] ;check for SDA high\n\t"
412  " ret ;if low return with false \n\t"
413  " sbis %[SCLIN],%[SCLPIN] ;check for SCL high \n\t"
414  " ret ;if low return with false \n\t"
415  " ldi r24,1 ;set return value to true \n\t"
416  " ret "
417  : :
418  [SCLDDR] "I" (SCL_DDR), [SCLPIN] "I" (SCL_PIN),
419  [SCLIN] "I" (SCL_IN), [SCLOUT] "I" (SCL_OUT),
420  [SDADDR] "I" (SDA_DDR), [SDAPIN] "I" (SDA_PIN),
421  [SDAIN] "I" (SDA_IN), [SDAOUT] "I" (SDA_OUT));
422  return true;
423 }
424 #endif // I2C_HARDWARE

426 bool i2c_start(uint8_t addr)
427 #if I2C_HARDWARE
428 {
429  uint8_t twst;
430 #if I2C_TIMEOUT
431  uint32_t start = millis();
432 #endif
433 
434  // send START condition
435  TWCR = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
436 
437  // wait until transmission completed
438  while(!(TWCR & (1<<TWINT))) {
439 #if I2C_TIMEOUT
440  if (millis() - start > I2C_TIMEOUT) return false;
441 #endif
442  }
443 
444  // check value of TWI Status Register. Mask prescaler bits.
445  twst = TW_STATUS & 0xF8;
446  if ( (twst != TW_START) && (twst != TW_REP_START)) return false;
447 
448  // send device address
449  TWDR = addr;
450  TWCR = (1<<TWINT) | (1<<TWEN);
451 
452  // wail until transmission completed and ACK/NACK has been received
453  while(!(TWCR & (1<<TWINT))) {
454 #if I2C_TIMEOUT
455  if (millis() - start > I2C_TIMEOUT) return false;
456 #endif
457  }
458 
459  // check value of TWI Status Register. Mask prescaler bits.
460  twst = TW_STATUS & 0xF8;
461  if ( (twst != TW_MT_SLA_ACK) && (twst != TW_MR_SLA_ACK) ) return false;
462 
463  return true;
464 }
465 #else // I2C_HARDWARE
466  {
467  __asm__ __volatile__
468  (
469 #if I2C_NOINTERRUPT
470  " cli ;clear IRQ bit \n\t"
471 #endif
472  " sbis %[SCLIN],%[SCLPIN] ;check for clock stretching slave\n\t"
473 #if __AVR_HAVE_JMP_CALL__
474  " call ass_i2c_wait_scl_high ;wait until SCL=H\n\t"
475 #else
476  " rcall ass_i2c_wait_scl_high ;wait until SCL=H\n\t"
477 #endif
478 #if I2C_PULLUP
479  " cbi %[SDAOUT],%[SDAPIN] ;disable pull-up \n\t"
480 #endif
481  " sbi %[SDADDR],%[SDAPIN] ;force SDA low \n\t"
482 #if __AVR_HAVE_JMP_CALL__
483  " call ass_i2c_delay_half ;wait T/2 \n\t"
484  " call ass_i2c_write ;now write address \n\t"
485 #else
486  " rcall ass_i2c_delay_half ;wait T/2 \n\t"
487  " rcall ass_i2c_write ;now write address \n\t"
488 #endif
489  " ret"
490  : : [SDADDR] "I" (SDA_DDR), [SDAPIN] "I" (SDA_PIN),
491  [SDAOUT] "I" (SDA_OUT), [SCLOUT] "I" (SCL_OUT),
492  [SCLIN] "I" (SCL_IN),[SCLPIN] "I" (SCL_PIN));
493  return true; // we never return here!
494 }
495 #endif // I2C_HARDWARE

562 bool i2c_start_wait(uint8_t addr)
563 #if I2C_HARDWARE
564 {
565  uint8_t twst;
566  uint16_t maxwait = I2C_MAXWAIT;
567 #if I2C_TIMEOUT
568  uint32_t start = millis();
569 #endif
570 
571  while (true) {
572  // send START condition
573  TWCR = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
574 
575  // wait until transmission completed
576  while(!(TWCR & (1<<TWINT))) {
577 #if I2C_TIMEOUT
578  if (millis() - start > I2C_TIMEOUT) return false;
579 #endif
580  }
581 
582  // check value of TWI Status Register. Mask prescaler bits.
583  twst = TW_STATUS & 0xF8;
584  if ( (twst != TW_START) && (twst != TW_REP_START)) continue;
585 
586  // send device address
587  TWDR = addr;
588  TWCR = (1<<TWINT) | (1<<TWEN);
589 
590  // wail until transmission completed
591  while(!(TWCR & (1<<TWINT))) {
592 #if I2C_TIMEOUT
593  if (millis() - start > I2C_TIMEOUT) return false;
594 #endif
595  }
596 
597  // check value of TWI Status Register. Mask prescaler bits.
598  twst = TW_STATUS & 0xF8;
599  if ( (twst == TW_MT_SLA_NACK )||(twst ==TW_MR_DATA_NACK) )
600  {
601  /* device busy, send stop condition to terminate write operation */
602  TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWSTO);
603 
604  // wait until stop condition is executed and bus released
605  while(TWCR & (1<<TWSTO)) {
606 #if I2C_TIMEOUT
607  if (millis() - start > I2C_TIMEOUT) return false;
608 #endif
609  }
610 
611  if (maxwait && --maxwait == 0) return false;
612 
613  continue;
614  }
615  //if( twst != TW_MT_SLA_ACK) return 1;
616  return true;
617  }
618 }
619 #else // I2C_HARDWARE
620  {
621  __asm__ __volatile__
622  (
623  " push r24 ;save original parameter \n\t"
624 #if I2C_MAXWAIT
625  " ldi r31, %[HIMAXWAIT] ;load max wait counter \n\t"
626  " ldi r30, %[LOMAXWAIT] ;load low byte \n\t"
627 #endif
628  "_Li2c_start_wait1: \n\t"
629  " pop r24 ;restore original parameter\n\t"
630  " push r24 ;and save again \n\t"
631 #if I2C_NOINTERRUPT
632  " cli ;disable interrupts \n\t"
633 #endif
634  " sbis %[SCLIN],%[SCLPIN] ;check for clock stretching slave\n\t"
635 #if __AVR_HAVE_JMP_CALL__
636  " call ass_i2c_wait_scl_high ;wait until SCL=H\n\t"
637 #else
638  " rcall ass_i2c_wait_scl_high ;wait until SCL=H\n\t"
639 #endif
640 #if I2C_PULLUP
641  " cbi %[SDAOUT],%[SDAPIN] ;disable pull-up \n\t"
642 #endif
643  " sbi %[SDADDR],%[SDAPIN] ;force SDA low \n\t"
644 #if __AVR_HAVE_JMP_CALL__
645  " call ass_i2c_delay_half ;delay T/2 \n\t"
646  " call ass_i2c_write ;write address \n\t"
647 #else
648  " rcall ass_i2c_delay_half ;delay T/2 \n\t"
649  " rcall ass_i2c_write ;write address \n\t"
650 #endif
651  " tst r24 ;if device not busy -> done \n\t"
652  " brne _Li2c_start_wait_done \n\t"
653 #if __AVR_HAVE_JMP_CALL__
654  " call ass_i2c_stop ;terminate write & enable IRQ \n\t"
655 #else
656  " rcall ass_i2c_stop ;terminate write & enable IRQ \n\t"
657 #endif
658 #if I2C_MAXWAIT
659  " sbiw r30,1 ;decrement max wait counter\n\t"
660  " breq _Li2c_start_wait_done ;if zero reached, exit with false -> r24 already zero!\n\t"
661 #endif
662 #if __AVR_HAVE_JMP_CALL__
663  " jmp _Li2c_start_wait1 ;device busy, poll ack again \n\t"
664 #else
665  " rjmp _Li2c_start_wait1 ;device busy, poll ack again \n\t"
666 #endif
667  "_Li2c_start_wait_done: \n\t"
668  " clr r25 ;clear high byte of return value\n\t"
669  " pop __tmp_reg__ ;pop off orig argument \n\t"
670  " ret "
671  : : [SDADDR] "I" (SDA_DDR), [SDAPIN] "I" (SDA_PIN), [SDAOUT] "I" (SDA_OUT),
672  [SCLIN] "I" (SCL_IN), [SCLPIN] "I" (SCL_PIN),
673  [HIMAXWAIT] "M" (I2C_MAXWAIT>>8),
674  [LOMAXWAIT] "M" (I2C_MAXWAIT&0xFF)
675  : "r30", "r31" );
676  return true; // fooling the compiler
677 }
